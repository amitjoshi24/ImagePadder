from pyodide.ffi import create_proxy import io import math import base64 from PIL import Image from js import document # Function to update the color preview def update_color_preview(event=None): color_input = document.getElementById("bg-color") color_value = color_input.value # Update the preview box preview = document.getElementById("color-preview") preview.style.backgroundColor = color_value # Update the hex text hex_text = document.getElementById("color-hex") hex_text.innerHTML = color_value.upper() # Set up the color picker event listener document.getElementById("bg-color").addEventListener("input", create_proxy(update_color_preview)) # Initialize the color preview update_color_preview() async def pyProcess(event=None): # Show loading spinner document.getElementById("loading-spinner").style.display = "block" # Clear previous error messages error_div = document.getElementById("error-message") error_div.innerHTML = "" # Get the selected file from the file input element file_input = document.getElementById("file-input") if file_input.files.length == 0: error_div.innerHTML = "Please select an image file." document.getElementById("loading-spinner").style.display = "none" return file = file_input.files.item(0) # Read the file as bytes using file.arrayBuffer() try: data = await file.arrayBuffer() image_bytes = bytes(bytearray(data.to_py())) except Exception as e: error_div.innerHTML = "Error reading file: " + str(e) document.getElementById("loading-spinner").style.display = "none" return # Open the image using Pillow from the bytes try: im = Image.open(io.BytesIO(image_bytes)) except Exception as e: error_div.innerHTML = "Image cannot be opened. " + str(e) document.getElementById("loading-spinner").style.display = "none" return # Check if transparent background is selected use_transparent_bg = document.getElementById("transparent-bg").checked # Get the background color from the color picker bg_color_hex = document.getElementById("bg-color").value # Convert hex to RGB bg_color_hex = bg_color_hex.lstrip('#') bg_color_rgb = tuple(int(bg_color_hex[i:i+2], 16) for i in (0, 2, 4)) # Determine which mode is active aspect_ratio_tab = document.getElementById("content-aspect-ratio") is_aspect_ratio_mode = aspect_ratio_tab.classList.contains("active") # Get the original image dimensions width, height = im.size if is_aspect_ratio_mode: # Aspect ratio mode aspect_ratio_input = document.getElementById("aspect-ratio").value try: nw, nh = tuple(int(num) for num in aspect_ratio_input.split('x')) except Exception as e: error_div.innerHTML = "Invalid aspect ratio. Must be in the format WIDTHxHEIGHT, e.g. 4x3." document.getElementById("loading-spinner").style.display = "none" return # Determine whether to pad vertically or horizontally vertical_pad = (nw / nh) <= (width / height) if vertical_pad: new_width = width new_height = int(math.ceil(width * nh / nw)) else: new_width = int(math.ceil(height * nw / nh)) new_height = height # When creating the new image, handle transparency if selected if use_transparent_bg: # Force RGBA mode for transparency if im.mode != "RGBA": im = im.convert("RGBA") # Use transparent background background_color = (255, 255, 255, 0) # Fully transparent result = Image.new("RGBA", (new_width, new_height), background_color) else: # Use selected background color if im.mode in ("RGBA", "LA"): background_color = bg_color_rgb + (255,) # Add alpha channel else: background_color = bg_color_rgb result = Image.new(im.mode, (new_width, new_height), background_color) # Calculate paste offset to center the image if vertical_pad: offset = ((new_height - height) // 2) result.paste(im, (0, offset)) else: offset = ((new_width - width) // 2) result.paste(im, (offset, 0)) else: # Custom padding mode try: padding_top = int(document.getElementById("padding-top").value) padding_right = int(document.getElementById("padding-right").value) padding_bottom = int(document.getElementById("padding-bottom").value) padding_left = int(document.getElementById("padding-left").value) except Exception as e: error_div.innerHTML = "Invalid padding values. Please enter positive numbers." document.getElementById("loading-spinner").style.display = "none" return # Calculate new dimensions new_width = width + padding_left + padding_right new_height = height + padding_top + padding_bottom # Paste the original image with the specified padding result.paste(im, (padding_left, padding_top)) # Save the result image to a bytes buffer in PNG format output_buffer = io.BytesIO() try: result.save(output_buffer, format="PNG") except Exception as e: error_div.innerHTML = "Error saving padded image: " + str(e) document.getElementById("loading-spinner").style.display = "none" return output_bytes = output_buffer.getvalue() # Encode the image to base64 so it can be embedded in the HTML b64_str = base64.b64encode(output_bytes).decode("utf-8") img_data = "data:image/png;base64," + b64_str # Hide loading spinner when done (whether successful or not) document.getElementById("loading-spinner").style.display = "none" # Hide placeholder and show image container document.getElementById("result-placeholder").style.display = "none" document.getElementById("image-container").style.display = "block" # Display the padded image result_img = document.getElementById("result-image") result_img.src = img_data # Update the download link with the new image data download_link = document.getElementById("download-link") download_link.href = img_data # Create a proxy for the PyScript function to ensure proper interaction document.getElementById("process-button").onclick = create_proxy(pyProcess) # Function to handle tab switching from PyScript def switch_tab(tab_id): # Get all tabs and contents tabs = document.querySelectorAll('.tab') contents = document.querySelectorAll('.tab-content') # Remove active class from all for i in range(tabs.length): tabs[i].classList.remove('active') for i in range(contents.length): contents[i].classList.remove('active') # Add active class to selected tab and content selected_tab = document.getElementById('tab-' + tab_id) selected_content = document.getElementById('content-' + tab_id) if selected_tab: selected_tab.classList.add('active') if selected_content: selected_content.classList.add('active') # Add click handlers for tabs document.getElementById("tab-aspect-ratio").addEventListener("click", create_proxy(lambda e: switch_tab('aspect-ratio'))) document.getElementById("tab-custom-padding").addEventListener("click", create_proxy(lambda e: switch_tab('custom-padding')))
