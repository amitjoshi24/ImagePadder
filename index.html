from pyodide.ffi import create_proxy import io import math import base64 from PIL import Image from js import document # Function to update the color preview def update_color_preview(event=None): color_input = document.getElementById("bg-color") color_value = color_input.value # Update the preview box preview = document.getElementById("color-preview") preview.style.backgroundColor = color_value # Update the hex text hex_text = document.getElementById("color-hex") hex_text.innerHTML = color_value.upper() # Set up the color picker event listener document.getElementById("bg-color").addEventListener("input", create_proxy(update_color_preview)) # Initialize the color preview update_color_preview() async def pyProcess(event=None): # Accept the event argument # Show loading spinner document.getElementById("loading-spinner").style.display = "block" # Clear previous error messages error_div = document.getElementById("error-message") error_div.innerHTML = "" # Get the selected file from the file input element file_input = document.getElementById("file-input") if file_input.files.length == 0: error_div.innerHTML = "Please select an image file." document.getElementById("loading-spinner").style.display = "none" return file = file_input.files.item(0) # Read the file as bytes using file.arrayBuffer() try: data = await file.arrayBuffer() # returns a promise; await it image_bytes = bytes(bytearray(data.to_py())) except Exception as e: error_div.innerHTML = "Error reading file: " + str(e) document.getElementById("loading-spinner").style.display = "none" return # Open the image using Pillow from the bytes try: im = Image.open(io.BytesIO(image_bytes)) except Exception as e: error_div.innerHTML = "Image cannot be opened. " + str(e) document.getElementById("loading-spinner").style.display = "none" return # Get and parse the aspect ratio input (e.g., "4x3") aspect_ratio_input = document.getElementById("aspect-ratio").value try: nw, nh = tuple(int(num) for num in aspect_ratio_input.split('x')) except Exception as e: error_div.innerHTML = "Invalid aspect ratio. Must be in the format WIDTHxHEIGHT, e.g. 4x3." document.getElementById("loading-spinner").style.display = "none" return # Get the background color from the color picker bg_color_hex = document.getElementById("bg-color").value # Convert hex to RGB bg_color_hex = bg_color_hex.lstrip('#') bg_color_rgb = tuple(int(bg_color_hex[i:i+2], 16) for i in (0, 2, 4)) # Get the original image dimensions width, height = im.size # Determine whether to pad vertically or horizontally vertical_pad = (nw / nh) <= (width / height) if vertical_pad: new_width = width new_height = int(math.ceil(width * nh / nw)) else: new_width = int(math.ceil(height * nw / nh)) new_height = height # Create a new image with the selected background color. Support transparency if needed. if im.mode in ("RGBA", "LA"): background_color = bg_color_rgb + (255,) # Add alpha channel else: background_color = bg_color_rgb result = Image.new(im.mode, (new_width, new_height), background_color) # Calculate paste offset to center the image if vertical_pad: offset = ((new_height - height) // 2) result.paste(im, (0, offset)) else: offset = ((new_width - width) // 2) result.paste(im, (offset, 0)) # Save the result image to a bytes buffer in PNG format output_buffer = io.BytesIO() try: result.save(output_buffer, format="PNG") except Exception as e: error_div.innerHTML = "Error saving padded image: " + str(e) document.getElementById("loading-spinner").style.display = "none" return output_bytes = output_buffer.getvalue() # Encode the image to base64 so it can be embedded in the HTML b64_str = base64.b64encode(output_bytes).decode("utf-8") img_data = "data:image/png;base64," + b64_str # Hide loading spinner when done (whether successful or not) document.getElementById("loading-spinner").style.display = "none" # Hide placeholder and show image container document.getElementById("result-placeholder").style.display = "none" document.getElementById("image-container").style.display = "block" # Display the padded image result_img = document.getElementById("result-image") result_img.src = img_data # Update the download link with the new image data download_link = document.getElementById("download-link") download_link.href = img_data # Create a proxy for the PyScript function to ensure proper interaction document.getElementById("process-button").onclick = create_proxy(pyProcess)
